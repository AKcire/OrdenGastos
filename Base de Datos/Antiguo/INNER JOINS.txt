TABLA CONTRATO - PROVEEDORES- TRABAJADORES

SELECT
E.UZFTEMPLOYEE_NOMBRE,
E.UZFTEMPLOYEE_CARGO,
P.UZFTPROVIDER_NOMBRE,
P.UZFTPROVIDER_RUC,
C.UZFTCONTRATO_DESCRIPCION,
C.UZFTCONTRATO_OBJETO_CONTRATA,
C.UZFTCONTRATO_NUMERO_CONTRATO
FROM UZFTCONTRATO C
INNER JOIN UZFTEMPLOYEE E ON C.UZFTEMPLOYEE_ID = E.UZFTEMPLOYEE_ID
INNER JOIN UZFTPROVIDER P ON C.UZFTPROVIDER_ID = P.UZFTPROVIDER_ID



TABLA CERTIFICADO _ CONTRATO

SELECT CE.UZFTCERTIFICATE_ID,
CE.UZFTCERTIFICATE_CODIGO_PRESUP,
CE.UZFTCERTIFICATE_DESCRIPCIONPRE,
CE.UZFTCERTIFICATE_VALOR,
C.UZFTCONTRATO_OBJETO_CONTRATA
FROM UZFTCERTIFICATE CE
INNER JOIN UZFTCONTRATO C ON CE.UZFTCONTRATO_ID = C.UZFTCONTRATO_ID



TABLA ORDEN GASTO - CONTRATO - CERTIFICADO- PROVEEDORES- TRABAJADORES

SELECT
E.UZFTEMPLOYEE_NOMBRE,
E.UZFTEMPLOYEE_CARGO,
P.UZFTPROVIDER_NOMBRE,
P.UZFTPROVIDER_RUC,
C.UZFTCONTRATO_DESCRIPCION,
C.UZFTCONTRATO_OBJETO_CONTRATA,
C.UZFTCONTRATO_NUMERO_CONTRATO,
CE.UZFTCERTIFICATE_CODIGO_PRESUP,
CE.UZFTCERTIFICATE_DESCRIPCIONPRE,
CE.UZFTCERTIFICATE_VALOR,
UZFTTYPEORDER.UZFTTYPEORDER_DESCRIPCION,
O.UZFTORDER_ID,
O.UZFTORDER_NUMERO,
O.UZFTORDER_FECHA,
O.UZFTORDER_MES,
O.UZFTORDER_CANTIDAD,
O.UZFTORDER_PRECIO_UNITARIO,
O.UZFTORDER_INSTALACION,
O.UZFTORDER_SUBTOTAL,
O.UZFTORDER_IVA,
O.UZFTORDER_OTRO_IMPUESTO,
O.UZFTORDER_TOTAL
FROM UZFTORDER O
INNER JOIN UZFTTYPEORDER ON O.UZFTTYPEORDER_ID = UZFTTYPEORDER.UZFTTYPEORDER_ID
INNER JOIN UZFTCONTRATO C ON O.UZFTCONTRATO_ID = C.UZFTCONTRATO_ID
INNER JOIN UZFTEMPLOYEE E ON E.UZFTEMPLOYEE_ID = C.UZFTEMPLOYEE_ID
INNER JOIN UZFTPROVIDER P ON P.UZFTPROVIDER_ID = C.UZFTPROVIDER_ID
INNER JOIN UZFTCERTIFICATE CE ON CE.UZFTCONTRATO_ID = C.UZFTCONTRATO_ID
